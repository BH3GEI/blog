### 1718. Construct the Lexicographically Largest Valid Sequence
```cpp
class Solution {
private:
    vector<int> res;
    vector<int> used;
    int m;
    
    bool dfs(int pos) {
        if (pos == res.size()) return true;
        if (res[pos] != 0) return dfs(pos + 1);
        
        for (int i = m; i >= 1; --i) {
            if (i == 1) {
                if (used[i] == 0) {
                    res[pos] = 1;
                    used[i] = 1;
                    if (dfs(pos + 1)) return true;
                    res[pos] = 0;
                    used[i] = 0;
                }
            } else {
                if (used[i] != 0) continue;
                int next_pos = pos + i;
                if (next_pos >= res.size() || res[next_pos] != 0) continue;
                
                res[pos] = i;
                res[next_pos] = i;
                used[i] = 2;
                if (dfs(pos + 1)) return true;
                
                res[pos] = 0;
                res[next_pos] = 0;
                used[i] = 0;
            }
        }
        return false;
    }
    
public:
    vector<int> constructDistancedSequence(int n) {
        m = n;
        int len = 2 * n - 1;
        res.resize(len, 0);
        used.resize(n + 1, 0);
        dfs(0);
        return res;
    }
};
```

### 1079. Letter Tile Possibilities
```cpp
class Solution {
public:
    int numTilePossibilities(string tiles) {
        // 统计字符频率
        unordered_map<char, int> freq;
        for (char c : tiles) {
            freq[c]++;
        }
        vector<int> counts;
        for (auto& p : freq) {
            counts.push_back(p.second);
        }
        
        // 预处理阶乘数组
        vector<int> fact(8, 1);
        for (int i = 1; i <= 7; ++i) {
            fact[i] = fact[i - 1] * i;
        }
        
        int result = 0;
        vector<int> current_selection(counts.size(), 0);
        backtrack(0, counts, current_selection, 0, result, fact);
        return result;
    }
    
private:
    void backtrack(int index, const vector<int>& counts, vector<int>& current_selection, int sum, int& result, const vector<int>& fact) {
        if (index == counts.size()) {
            if (sum == 0) return;
            // 计算排列数目
            int perm = fact[sum];
            for (int num : current_selection) {
                perm /= fact[num];
            }
            result += perm;
            return;
        }
        
        int max_choose = counts[index];
        for (int choose = 0; choose <= max_choose; ++choose) {
            current_selection[index] = choose;
            backtrack(index + 1, counts, current_selection, sum + choose, result, fact);
        }
    }
};
```
